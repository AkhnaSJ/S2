
-- shows all shells available on system
$ cat /etc/shells

-- path of bash shell
$ which bash

$ ls -al

-- set permissions(executable : +x)
chmod +x filename.sh

-- to create a new file
$ touch filename.sh(txt)
$ nano filename.sh(txt)

-- run
$ ./filename.sh (set executable permissions)
$ sh filename.sh
$ bash filename.sh
